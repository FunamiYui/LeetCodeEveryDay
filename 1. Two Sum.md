### Brute Force O(n^2)
```c++
class Solution {
public:
    vector<int> twoSum(vector<int>& nums, int target) {
        int i, j;
        bool flag = false;
        vector<int> ans;
        for (int i = 0; i < nums.size(); i++) {
            for (int j = 0; j < nums.size(); j++) {
                if (nums[i] + nums[j] == target) {
                    if (i != j) {
                        ans.push_back(i);
                        ans.push_back(j);
                        flag = true;
                        break;
                    }
                }
            }
            if (flag) {
                break;
            }
        }
        return ans;
    }
};
```
